package co.com.bnpparibas.cardif.closingclaims.api.exceptionhandler;

import co.com.bnpparibas.cardif.closingclaims.domain.dtos.response.error.ErrorDetail;
import co.com.bnpparibas.cardif.closingclaims.domain.dtos.response.error.ErrorHeader;
import co.com.bnpparibas.cardif.closingclaims.domain.dtos.response.error.ResponseErrorModel;
import co.com.bnpparibas.cardif.closingclaims.domain.util.exception.BusinessException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;

import java.util.Date;
import java.util.Objects;
import java.util.UUID;
import java.util.stream.Collectors;

/**
 * Global exception handler for the REST controllers.
 * <p>
 * This class intercepts exceptions thrown by controllers and translates them
 * into a {@link ResponseErrorModel} wrapped in a {@link ResponseEntity}
 * with the appropriate HTTP status.
 * </p>
 */
@ControllerAdvice
public class ControllerExceptionHandler {

    private static final Logger log = LoggerFactory.getLogger(ControllerExceptionHandler.class);

    /**
     * Handles all {@link BusinessException}
     * instances thrown from controller methods.
     *
     * @param e the {@link BusinessException} captured
     * @return a {@link ResponseEntity}
     * containing a {@link ResponseErrorModel}
     * with details of the error and the HTTP status defined in the exception
     */
    @ExceptionHandler(BusinessException.class)
    public ResponseEntity<ResponseErrorModel> handleBusinessException(BusinessException e) {

        log.error("Error: BusinessException", e);

        return createResponse(
                e.getHttpStatus().value(),
                e.getHttpStatus().getReasonPhrase(),
                e.getBusinessCode(),
                e.getBusinessMessage(),
                Objects.nonNull(e.getEx()) ? e.getEx() : e,
                e.getHttpStatus()
        );
    }

    /**
     * Handles request validation errors generated by {@code @Valid}.
     *
     * @param e the {@link MethodArgumentNotValidException} captured
     * @return a {@link ResponseEntity} containing validation error details
     */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ResponseErrorModel> handleValidationException(
            MethodArgumentNotValidException e) {

        log.error("Error: MethodArgumentNotValidException", e);

        String validationMessages = e.getBindingResult()
                .getFieldErrors()
                .stream()
                .map(error -> error.getField() + ": " + error.getDefaultMessage())
                .collect(Collectors.joining(", "));

        return createResponse(
                HttpStatus.BAD_REQUEST.value(),
                HttpStatus.BAD_REQUEST.getReasonPhrase(),
                "VALIDATION_ERROR",
                validationMessages,
                e,
                HttpStatus.BAD_REQUEST
        );
    }

    /**
     * Builds the {@link ResponseEntity} that contains the {@link ResponseErrorModel}
     * to be returned to the client.
     *
     * @param returnCode    the numeric return code (typically the HTTP status code)
     * @param message       the short message associated with the return code
     * @param code          the business-specific error code
     * @param detailMessage the detailed error description
     * @param e             the exception that caused the error
     * @param status        the HTTP status to be set in the response
     * @return a {@link ResponseEntity} with the populated {@link ResponseErrorModel}
     */
    private ResponseEntity<ResponseErrorModel> createResponse(int returnCode,
                                                              String message,
                                                              String code,
                                                              String detailMessage,
                                                              Exception e,
                                                              HttpStatus status) {
        return new ResponseEntity<>(ResponseErrorModel
                .builder()
                .correlationId(String.valueOf(UUID.randomUUID()))
                .errorDetail(ErrorDetail.builder()
                        .code(code)
                        .errorDate(new Date())
                        .errorStack(e.getLocalizedMessage())
                        .message(detailMessage)
                        .build())
                .errorHeader(ErrorHeader.builder()
                        .returnCode(returnCode)
                        .message(message)
                        .build())
                .build(),
                status);
    }
}
